1    import os

class dpro():
    path = "../Data_from_Goliath"
    directory = os.listdir(path)  # returns list

    fh = open("data.csv", "w")
    for filename in directory:
        try:
            file = path + "/" + filename
            f = open(file)
            count = -1
            process = False
            real = 0
            user = 0
            sys = 0
            for line in f:
                count = count + 1;

                if(count == 2):
                    process = True
                    words = line.split();
                    if(words.__len__() == 3):
                        if(words[0] == "[N,D]"):
                            ND = words[2]
                            ND = ND.replace("[", "")
                            ND = ND.replace("]", "")
                            ND = ND.split(",")
                            N = ND[0]
                            D = ND[1]

                if(process and count == 6):
                    words = line.split();
                    temp = words[1]
                    temp = temp.replace("s", "")
                    temp = temp.split("m")
                    real = 60*int(temp[0])
                    temp = temp[1]
                    temp = temp.split(".")
                    real = real + int(temp[0]) + (int(temp[1])/1000)

                if(process and count == 7):
                    words = line.split();
                    temp = words[1]
                    temp = temp.replace("s", "")
                    temp = temp.split("m")
                    user = 60*int(temp[0])
                    temp = temp[1]
                    temp = temp.split(".")
                    user = user + int(temp[0]) + (int(temp[1])/1000)

                if(process and count == 8):
                    words = line.split();
                    temp = words[1]
                    temp = temp.replace("s", "")
                    temp = temp.split("m")
                    sys = 60*int(temp[0])
                    temp = temp[1]
                    temp = temp.split(".")
                    sys = sys + int(temp[0]) + (int(temp[1])/1000)


            if (process) and ((real != 0) and (user != 0) and (sys != 0)):
                fh.write(str(N) + "," + str(D) + "," +str(real) + "," +str(user) + "," +str(sys) + "\n")
        except:
            pass
    fh.close
